# append() - Add element at end
arr.append(4)        # [1,2,3] → [1,2,3,4]

# pop() - Remove last element
arr.pop()            # [1,2,3,4] → [1,2,3]

# pop(i) - Remove at index
arr.pop(1)           # [1,2,3] → [1,3]

# insert(i, val) - Add at specific index
arr.insert(1, 99)    # [1,2,3] → [1,99,2,3]

# remove(val) - Finds first occurrence and removes it
arr.remove(2)        # [1,2,3,2] → [1,3,2]
